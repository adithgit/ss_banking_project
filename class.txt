classDiagram
    direction LR
    note "Represents C structs and related functions"

    class Admin {
        <<Role>>
        +handleAdminSession()
        +createNewEmployee()
        +modifyUser()
        +updateEmployeeRole()
        +changeAdminPassword()
    }

    class Manager {
        <<Role>>
        +handleManagerSession()
        +authenticateManager()
        +setAccountActiveStatus()
        +assignLoanToEmployee()
        +reviewClientFeedback()
        +updateManagerPassword()
    }

    class Employee {
        <<struct>>
        employeeID: int
        firstName: string
        lastName: string
        password: string
        roleType: int
        <<Functions>>
        +handleEmployeeSession()
        +authenticateEmployee()
        +createNewCustomerAccount()
        +processLoanApplication()
        +viewAssignedLoans()
        +updateEmployeePassword()
        +modifyUser()  // Calls admin func
        +viewTransactionLogs() // Calls customer func
    }

    class AccountHolder {
        <<struct>>
        accountID: int
        currentBalance: float
        holderName: string
        password: string
        isActive: int
        <<Functions>>
        +handleCustomerSession()
        +authenticateCustomer()
        +processWithdrawal()
        +processDeposit()
        +checkBalance()
        +requestLoan()
        +executeTransfer()
        +submitFeedback()
        +viewTransactionLogs()
        +updateCustomerPassword()
        +endUserSession() // Shared logout
    }

    class LoanRecord {
        <<struct>>
        assignedEmployeeID: int
        accountID: int
        loanRecordID: int
        amount: int
        loanStatus: int
    }

    class TransactionLog {
        <<struct>>
        accountID: int
        logEntry: string
    }

    class ClientFeedback {
        <<struct>>
        message: string
    }

    class IDGenerator {
        <<struct>>
        nextID: int
    }
